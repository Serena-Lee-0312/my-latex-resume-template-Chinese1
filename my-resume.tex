        #1 & \textit{\footnotesize{#3}} \\   [-4pt]% 改成负值，向上拉近
        \multicolumn{2}{@{}p{\linewidth}@{}}{%
            \footnotesize{#2}%
        } \\
    \end{tabularx}
}

\newcommand{\subHeading}[4]{%
  \item
  \begin{tabularx}{0.98\textwidth}[t]{@{}X r@{}}  
    \textbf{#1} & \textit{\footnotesize{#4}} \\[-4pt]% 改成负值，向上拉近
    \footnotesize #3 & \textit{\footnotesize #2} \\
  \end{tabularx}
}

\newcommand{\itemStart}{               % 定义新命令
  \begin{justify}                                % 开始两端对齐环境
    \begin{itemize}[                              % 开始无序列表，带可选参数
      leftmargin=3ex,      % 列表左缩进 3ex（相对于当前字体 x 高度）
      rightmargin=2ex,     % 列表右缩进 2ex
      noitemsep,           % 取消列表项之间的额外间距
      labelsep=1.2mm,      % 标签（项目符号）与内容之间的间距 1.2mm
    ]
    \footnotesize
    \kaishu %字体设置，楷书
    \vspace{-3.58mm}
}
